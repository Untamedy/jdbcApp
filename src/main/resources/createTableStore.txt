CREATE TABLE Goods  (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,name VARCHAR(30) NOT NULL,articul INT(30) NOT NULL
CREATE TABLE Clients (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,name VARCHAR(30) NOT NULL,phoneNum VARCHAR(12) NOT NULL
CREATE TABLE Orders (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,code VARCHAR(30) NOT NULL,client_id INT(30) NOT NULL
CREATE TABLE Orders_goods (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,order_id INT(30) NOT NULL,goods_id INT(30) NOT NULL
